 51    def _get_unique_list(self, seq: List[List[str]]) -> List[List[str]]:
 52        seen = []
 53        return [x for x in seq if x not in seen and not seen.append(x)]
 54
 55    # [['i'], ['i','l', 'j'], ['j','k']] => ['l','k']
 56    def _compute_output_template(self, template: List[List[str]]) -> List[Union[str, Any]]:
 57        counter = collections.Counter(chain.from_iterable(template))
 58        out_template = [k for k, cnt in counter.items() if cnt == 1 and k != "b"]
 59        return sorted(out_template)

 61    # [['i'], ['i','l', 'j'], ['j','k']] => ['l','k']
 62    # [[3], [3, 4, 5], [5,6]] => [4,6]
 63    def _compute_output_shape(self, out_template: List[Union[str, Any]], sw_node_template: List[List[str]], sw_node_shape: List[Union[Tuple[int, int], List[int], List[Optional[int]], Tuple[int]]]) -> List[Union[int, Any]]:
 64        symbol_shape = {}
 65        for template_list, shape_list in zip(sw_node_template, sw_node_shape):
 66            for t, s in zip(template_list, shape_list):
 67                if t not in symbol_shape:
 68                    symbol_shape[t] = s
 69                elif symbol_shape[t] is None:
 70                    symbol_shape[t] = s
 71                else:
 72                    assert symbol_shape[t] == s, (
 73                        "index symbol mismatch:"
 74                        + str(t)
 75                        + ":"
 76                        + str(symbol_shape[t])
 77                        + "!="
 78                        + str(s)
 79                    )
 80        out_shape = []
 81        for symbol in out_template:
 82            if symbol in symbol_shape:
 83                out_shape.append(symbol_shape[symbol])
 84            else:
 85                out_shape.append(None)
 86        return out_shape

 88    def _unify_shapes(self, path_shapes: List[List[Union[int, Any]]]) -> List[Union[int, Any]]:
 89        """
 90        This method is used to unify shapes for all paths
 91        """
 92        n = len(path_shapes)
 93        if n == 0:
 94            return []
 95        else:
 96            m = len(path_shapes[0])
 97            out_shape = []
 98            for j in range(m):
 99                dim = None
100                for i in range(n):
101                    if path_shapes[i][j] is None:
102                        pass
103                    elif dim is None:
104                        dim = path_shapes[i][j]
105                    else:
106                        assert path_shapes[i][j] == dim, "shape mismatching"
107                out_shape.append(dim)
108            return out_shape

110    def build_explanation_graph_template(
111        self, graph, tensor_provider, operator_loader=None, cycle_node=[]
112    ):
113        """
114        Args:
115            graph: explanation graph object
116            tensor_provider (SwitchTensorProvider): tensor provider
117            operator_loader (OperatorLoader): operator loader
118            cycle_node (List[int]): a list of sorted_id of cycle nodes
119
120        Returns:
121            A tuple containing goal_template and cycle_node
122             - goal_template (List[Dict[str, Any]]): a list of goal templates
123             - cycle_node (List[int]): a list of cycle node (if given cycle_node, it is updated)
124        """
125        # checking template
126        goal_template = [None] * len(graph.goals)
127        for i in range(len(graph.goals)):
128            g = graph.goals[i]
129            path_template = []
130            path_shape = []
131            path_batch_flag = False
132            for path in g.paths:
133                ## build template and inside for switches in the path
134                sw_template = []
135                sw_shape = []
136                for sw in path.tensor_switches:
137                    ph = tensor_provider.get_placeholder_name(sw.name)
138                    sw_obj = tensor_provider.get_switch(sw.name)
139                    if len(ph) > 0:
140                        sw_template.append(["b"] + list(sw.values))
141                        path_batch_flag = True
142                        sw_shape.append([None] + list(sw_obj.get_shape()))
143                    else:
144                        sw_template.append(list(sw.values))
145                        sw_shape.append(sw_obj.get_shape())
146                ## building template and inside for nodes in the path
147                node_template = []
148                node_shape = []
149                cycle_detected = False
150                for node in path.nodes:
151                    temp_goal = goal_template[node.sorted_id]
152                    if temp_goal is None:
153                        # cycle
154                        if node.sorted_id not in cycle_node:
155                            cycle_node.append(node.sorted_id)
156                        cycle_detected = True
157                        continue
158                    if len(temp_goal["template"]) > 0:
159                        if temp_goal["batch_flag"]:
160                            path_batch_flag = True
161                        node_shape.append(temp_goal["shape"])
162                        node_template.append(temp_goal["template"])
163                if cycle_detected:
164                    continue
165                sw_node_template = sw_template + node_template
166                sw_node_shape = sw_shape + node_shape
167
168                ##########
169                ops = [op.name for op in path.operators]
170                if "distribution" in ops:
171                    # distributino clause
172                    print("=== distribution ===")
173                    print(sw_node_template)
174                    print(sw_node_shape)
175                    ##
176                    out_template = sw_node_template[0]
177                    out_shape = sw_node_shape[0]
178                    print(out_template)
179                    print(out_shape)
180                else:
181                    # constructing einsum operation using template and inside
182                    out_template = self._compute_output_template(sw_node_template)
183                    out_shape = self._compute_output_shape(
184                        out_template, sw_node_template, sw_node_shape
185                    )
186                    if len(sw_node_template) > 0:  # condition for einsum
187                        if path_batch_flag:
188                            out_template = ["b"] + out_template
189                    ## computing operaters
190                    for op in path.operators:
191                        cls = operator_loader.get_operator(op.name)
192                        op_obj = cls(op.values)
193                        out_template = op_obj.get_output_template(out_template)
194                ##########
195                path_template.append(out_template)
196                path_shape.append(out_shape)
197                ##
198            ##
199            path_template_list = self._get_unique_list(path_template)
200            path_shape = self._unify_shapes(path_shape)
201            if len(path_template_list) == 0:
202                goal_template[i] = {
203                    "template": [],
204                    "batch_flag": False,
205                    "shape": path_shape,
206                }
207            else:
208                if len(path_template_list) != 1:
209                    print("[WARNING] missmatch indices:", path_template_list)
210                goal_template[i] = {
211                    "template": path_template_list[0],
212                    "batch_flag": path_batch_flag,
213                    "shape": path_shape,
214                }
215        ##
216        return goal_template, cycle_node

219class SwitchTensor:
220    """ This class connect a tensor with a switch
221    
222    Attributes:
223        name (str): switch name
224        shape_set (Set[Tuple[int,...]]): a set of tensor shapes associated with this switch
225        ph_names (List[str]): generated from the switch name
226        vocab_name (str): generated from the switch name
227        var_name (str): generated from the switch name
228        value (Any): this values is used to represent x in the special tensor atom: get(_,x)
229
230    """
231    def __init__(self, sw_name: str) -> None:
232        self.value = None
233        self.name = sw_name
234        self.shape_set = set([])
235        self.ph_names = self.get_placeholder_name(sw_name)
236        self.vocab_name = self.make_vocab_name(sw_name) # update self.value
237        self.var_name = self.make_var_name(sw_name)
238
239    def enabled_placeholder(self):
240        return len(self.ph_names) == 0
241
242    def add_shape(self, shape: Tuple[int,...]) -> None:
243        self.shape_set.add(shape)
244
245    def get_shape(self) -> Tuple[int,...]:
246        assert len(self.shape_set) == 1, (
247            self.name + ": shape is not unique:" + str(self.shape_set)
248        )
249        return list(self.shape_set)[0]
250
251    def get_placeholder_name(self, name: str) -> List[Union[str, Any]]:
252        pattern = r"(\$placeholder[0-9]+\$)"
253        m = re.finditer(pattern, name)
254        names = [el.group(1) for el in m]
255        return names
256
257    def make_vocab_name(self, name: str) -> str:
258        m = re.match(r"^tensor\(get\((.*),([0-9]*)\)\)$", name)
259        if m:
260            name = "tensor(" + m.group(1) + ")"
261            self.value = int(m.group(2))
262        pattern = r"\$(placeholder[0-9]+)\$"
263        m = re.sub(pattern, "", name)
264        return self.make_var_name(m)
265
266    def make_var_name(self, name: str) -> str:
267        return re.sub(r"[\[\],\)\(\'$]+", "_", name)

271    """ This class connect a value object with a vovabrary via placeholders
272
273    vocab -> placeholder ->values
274
275    Attributes:
276        vocab_values (Dict[str, List[Any]]): the key is a vocab name, and the value ia a list of values.
277        value_index (Dict[Tuple[str,Any],int): the key is a tuple of a vocab name and a value, and the value is an index.
278
279    """
280
281    def __init__(self) -> None:
282        # vocab name => a list of values
283        self.vocab_values = None
284        # vocab name, value => index

287    def build_from_ph(self, ph_graph: 'PlaceholderGraph') -> None:
288        """ This method builds vocab_values and value_index from PlaceholderGraph.
289        """
290        vocab_ph = ph_graph.vocab_ph
291        ph_values = ph_graph.ph_values #
292        vocab_values = {}
293        for vocab_name, phs in vocab_ph.items():
294            for ph in phs:
295                if vocab_name not in vocab_values:
296                    vocab_values[vocab_name] = set()
297                vocab_values[vocab_name] |= ph_values[ph]
298        self.vocab_values = {k: list(v) for k, v in vocab_values.items()}
299        self.value_index = self._build_value_index()
300
301    def _build_value_index(self) -> Dict[Tuple[str, int32], int]:
302        value_index = {}
303        for vocab_name, values in self.vocab_values.items():
304            for i, v in enumerate(sorted(values)):
305                value_index[(vocab_name, v)] = i
306        return value_index
307
308    def get_values_index(self, vocab_name: str, value: Union[int, int32]) -> int:
309        key = (vocab_name, value)
310        if key in self.value_index:
311            return self.value_index[key]
312        else:
313            return 0
314
315    def get_values(self, vocab_name: str) -> Optional[List[int32]]:
316        if vocab_name not in self.vocab_values:
317            return None
318        return self.vocab_values[vocab_name]

321class PlaceholderGraph:
322    """ This class build a graph related to placeholders
323
324    vocab <-- sw_info --> placeholder --> values
325
326    Attributes:
327        vocab_ph (Dict[str, Set[str]]): vocab_name => a set of nemes of placeholders
328        ph_vocab (Dict[str, Set[str]]): placeholder name => a set of the vocabs
329        ph_values (Dict[str,Set[Any]]): placeholder name => a set of values
330        vocab_shape (Dict[str, Set[Tuple[int,...]]]): vocab_name => a set of shapes
331
332    """
333
334    def __init__(self) -> None:
335        self.vocab_ph = None
336        self.ph_vocab = None
337        self.ph_values = None #Dict[str,Set[Any]]
338        self.vocab_shape = None

340    def _build_ph_values(self, input_data: List[Dict[str, Union[int, List[str], ndarray]]]) -> None:
341        ph_values = {}
342        for g in input_data:
343            for ph in g["placeholders"]:
344                if ph not in ph_values:
345                    ph_values[ph] = set()
346            placeholders = [ph for ph in g["placeholders"]]
347            rt = np.transpose(g["records"])
348            for i, item in enumerate(rt):
349                ph_values[placeholders[i]] |= set(item)
350        self.ph_values = ph_values
351
352    def _build_vocab_ph(self, ph_values: Dict[str, Set[int32]], sw_info: Dict[str, SwitchTensor]) -> None:
353        # ph_vocab/vocab_ph: ph_name <== sw_info ==> vocab_name
354        # vocab_shape: vocab_name => shape
355        ph_vocab = {ph_name: set() for ph_name in ph_values.keys()}
356        vocab_ph = {sw.vocab_name: set() for sw in sw_info.values()}
357        vocab_shape = {sw.vocab_name: set() for sw in sw_info.values()}
358        for sw_name, sw in sw_info.items():
359            ## build vocab. shape
360            if sw.vocab_name not in vocab_shape:
361                vocab_shape[sw.vocab_name] = set()
362            vocab_shape[sw.vocab_name] |= sw.shape_set
363            ## build ph_vocab/vocab_ph
364            ph_list = sw.ph_names
365            if len(ph_list) == 1:
366                vocab_ph[sw.vocab_name].add(ph_list[0])
367                ph_vocab[ph_list[0]].add(sw.vocab_name)
368            elif len(ph_list) > 1:
369                print("[ERROR] not supprted: one placeholder for one term")
370        self.ph_vocab = ph_vocab
371        self.vocab_ph = vocab_ph
372        self.vocab_shape = vocab_shape
373        ##
374
375    def build(self, input_data: List[Dict[str, Union[int, List[str], ndarray]]], sw_info: Dict[str, SwitchTensor]) -> None:
376        if input_data is not None:
377            self._build_ph_values(input_data)
378        else:
379            self.ph_values = {}
380        self._build_vocab_ph(self.ph_values, sw_info)

564    def build(
565        self,
566        graph,
567        options,
568        input_data,
569        flags,
570        load_embeddings=False,
571        embedding_generators=[],
572        verbose=False,
573    ):
574        """
575        As a preparation before creating a computational graph, associate switches and tensors
576        
577         1. building PlaceholderGraph
578         2. building/loading VocabSet
579         3. associating VocabSet with variable types such as dataset, constant one-hot, and variables.
580         4. building placeholders
581         5. associating switches with tensors using embedding generators
582        
583        As a result, vocab_var, ph_var, and tensor_embedding in this class are constructed
584        """
585        # sw_info: switch name =>SwitchTensor
586        sw_info = self._build_sw_info(graph, options)
587        #
588        ph_graph = PlaceholderGraph()
589        ph_graph.build(input_data, sw_info)
590
591        ## build vocab group
592        if load_embeddings:
593            print("[LOAD]", flags.vocab)
594            with open(flags.vocab, mode="rb") as f:
595                vocab_set = pickle.load(f)
596        else:
597            vocab_set = VocabSet()
598            vocab_set.build_from_ph(ph_graph)
599            print("[SAVE]", flags.vocab)
600            with open(flags.vocab, mode="wb") as f:
601                pickle.dump(vocab_set, f)
602        ##
603        self.vocab_var_type = self._build_vocab_var_type(
604            ph_graph, vocab_set, embedding_generators
605        )
606        self.vocab_set = vocab_set
607        self.ph_graph = ph_graph
608        self.sw_info = sw_info
609        ##
610        # build placeholders
611        # ph_var    : ph_name => placeholder
612        ph_var = {}
613        batch_size = flags.sgd_minibatch_size
614        for ph_name in ph_graph.ph_values.keys():
