 49class TprismEvaluator:
 50    """Evaluator for pytorch system
 51
 52    """
 53    def __init__(self, goal_dataset=None):
 54        if goal_dataset is not None:
 55            self.n_goals = len(goal_dataset)
 56        else:
 57            self.n_goals = 1
 58        self.loss_history = [[] for _ in range(self.n_goals)]
 59        self.loss_dict_history = [[] for _ in range(self.n_goals)]
 60        self.label = [[] for _ in range(self.n_goals)]
 61        self.output = [[] for _ in range(self.n_goals)]
 62
 63    def start_epoch(self):
 64        self.running_loss = [0.0 for _ in range(self.n_goals)]
 65        self.running_loss_dict = [{} for _ in range(self.n_goals)]
 66        self.running_count = [0 for _ in range(self.n_goals)]
 67
 68    def update(self, loss, loss_dict, j):
 69        """
 70            This function is called the last of batch iteration
 71              loss,output, label: return values of forward
 72              j: goal index of iteration
 73              num_itr:: number of iterations
 74        """
 75        self.running_loss[j] += loss
 76        self.running_count[j] += 1
 77        for k, v in loss_dict.items():
 78            if k in self.running_loss_dict:
 79                self.running_loss_dict[j][k] += v
 80            else:
 81                self.running_loss_dict[j][k] = v
 82
 83    def stop_epoch(self, j=0, mean_flag=True):
 84        if mean_flag:
 85            self.running_loss[j] /= self.running_count[j]
 86            for k in self.running_loss_dict[j].keys():
 87                self.running_loss_dict[j][k] /= self.running_count[j]
 88        self.loss_history[j].append(self.running_loss[j])
 89        self.loss_dict_history[j].append(self.running_loss_dict[j])
 90
 91    def get_dict(self, prefix="train"):
 92        result = {}
 93        for j in range(self.n_goals):
 94            key = "{:s}-loss".format(prefix)
 95            val = self.running_loss[j]
 96            result[key] = float(val)
 97            for k, v in self.running_loss_dict[j].items():
 98                if k[0] != "*":
 99                    m = "{:s}-{:s}-loss".format(prefix, k)
100                else:
101                    m = "*{:s}-{:s}".format(prefix, k[1:])
102                result[m] = float(v)
103            return result
104
105    def get_msg(self, prefix="train"):
106        msg = []
107        for key, val in self.get_dict(prefix=prefix).items():
108            m = "{:s}: {:.3f}".format(key, val)
109            msg.append(m)
110        return "  ".join(msg)
111
112    def get_loss(self):
113        return self.running_loss
114
115    def update_data(self, output, label, j):
116        if type(output[j]) == list:  # preference
117            _o = [o.detach().numpy() for o in output[j]]
118        else:
119            _o = output[j].detach().numpy()
120        self.output[j].extend(_o)
121        if label is not None:
122            _l = label[j].detach().numpy()
123            self.label[j].extend(_l)

384            train_evaluator.stop_epoch()
